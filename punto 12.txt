DIM VEC(15) AS INTEGER
PRINT "INGRESE 15 NUMEROS"
DO
    FOR I = 1 TO 15
        INPUT VEC(I)
        PRINT "INGRESO"; I; "NUMEROS"
        SUMA = SUMA + VEC(I)

    NEXT
LOOP UNTIL NOT VEC(I) = 0
PROMEDIO = SUMA / 15

PRINT "ESTA ES LA SUMA"; SUMA
PRINT "ESTE ES EL PROMEDIO"; PROMEDIO
PRINT "ESTOS SON LOS NUMERO INGRESADOS"
FOR I = 1 TO 15
    PRINT VEC(I)
NEXT

DIM VEC(12) AS INTEGER
PRINT "INGRESE 12 NUMEROS"
FOR I = 1 TO 12
    INPUT VEC(I)
    PRINT "INGRESO"; I; "NUMEROS"
    IF VEC(I) MOD 2 = 0 THEN
        SUMAPARES = SUMAPARES + VEC(I)
        IF VEC(I) MOD 2 = 1 THEN
            SUMAIMP = SUMAIMP + VEC(I)
        END IF
    END IF
NEXT
PRINT "LOS NUMEROS PARES SON"
FOR I = 1 TO 12
    IF VEC(I) MOD 2 = 0 THEN
        PRINT VEC(I)

        FOR I = 1 TO 12
            PRINT "LOS NUMEROS IMPARES SON"
            IF VEC(I) MOD 2 = 1 THEN
                PRINT VEC(I)
            END IF
        NEXT
    END IF
NEXT



PRINT "LA SUMA DE LOS NUMEROS IMPARES SON"; SUMAIMP
PRINT "LA SUMA DE LOS NUMEROS PARES SON"; SUMAPARES
